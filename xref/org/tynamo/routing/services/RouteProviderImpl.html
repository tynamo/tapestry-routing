1   package org.tynamo.routing.services;
2   
3   import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
4   import org.tynamo.routing.Route;
5   
6   import java.util.Collections;
7   import java.util.List;
8   import java.util.Map;
9   
10  public class RouteProviderImpl implements RouteProvider {
11  
12  	private final List<Route> routes;
13  	private final Map<String, Route> routeMap = CollectionFactory.newConcurrentMap();
14  
15  	public RouteProviderImpl(List<Route> routes) {
16  		this.routes = Collections.unmodifiableList(routes);
17  		for (Route route : routes) {
18  			routeMap.put(route.getCanonicalizedPageName(), route);
19  		}
20  	}
21  
22  	@Override
23  	public Route getRoute(String canonicalizedPageName) {
24  		return routeMap.get(canonicalizedPageName);
25  	}
26  
27  	@Override
28  	public List<Route> getRoutes() {
29  		return routes;
30  	}
31  }
