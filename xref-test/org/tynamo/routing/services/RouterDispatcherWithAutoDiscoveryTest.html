17  	@Test
18  	public void auto_discovery_enabled() {
19  
20  		Registry registry = getRegistry(TapestryModule.class, RoutingModule.class, TestsModule.class);
21  
22  		AnnotatedPagesManager provider = registry.getService(AnnotatedPagesManager.class);

24  
25  		registry.cleanupThread();
26  		registry.shutdown();
27  
28  	}

30  	@Test
31  	public void auto_discovery_disabled_only_one_contributed_service() {
32  
33  		Registry registry = getRegistry(TapestryModule.class, RoutingModule.class, AutoDiscoveryDisabledModule.class);
34  
35  		AnnotatedPagesManager provider = registry.getService(AnnotatedPagesManager.class);
36  		Assert.assertEquals(provider.getRoutes().size(), 1, "there is one contributed page and autodiscovery is disabled");
37  
38  		registry.cleanupThread();
39  		registry.shutdown();
40  
41  	}

43  	private Registry getRegistry(Class... moduleClasses) {
44  
45  		RegistryBuilder builder = new RegistryBuilder();
46  		builder.add(moduleClasses);
47  		Registry registry = builder.build();
48  
49  		ApplicationGlobals globals = registry.getObject(ApplicationGlobals.class, null);
50  		globals.storeContext(new PageTesterContext("src/test/webapp"));
51  
52  		registry.performRegistryStartup();
53  		return registry;
54  	}
